1. 并查集

class Set{
	private:
		int fa[100000];//父亲 
		int siz[100000];//大小
	public:
		int Find(int x)
		{
			if(fa[x]==x)return x;
			else return fa[x]=Find(fa[x]);
		} 
		void Union(int u,int v)
		{
			int a=Find(u),b=Find(v),t;
			if(siz[a]>siz[b])t=a,a=b,b=t;
		    fa[a]=b,siz[a]+=siz[b];
		}
		void prer(int n)
		{
			for(int p=1;p<=n;p++)
				fa[p]=p;
		} 
};
